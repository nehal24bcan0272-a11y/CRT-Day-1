// Problem 1 
// Program to convert celcius to Fahrenheit

#include <iostream>
using namespace std;

int main() {
    double celcius;
    double fahrenheit;
    cin >> celcius;
    
    fahrenheit = (celcius * 9.0 / 5.0) + 32;
    cout << fahrenheit << endl;
    return 0;
}
